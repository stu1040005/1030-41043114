日歷展示

fullcalendar.js 官網

範例程式碼


            <!-- fullcalendar.js -->
            <script src="https://cdn.jsdelivr.net/npm/fullcalendar@6.1.15/index.global.min.js"></script>
            <script>
            document.addEventListener('DOMContentLoaded', function () {
                // 獲取日曆容器
                let calendarEl = document.getElementById('calendar');
          
                // 初始化 FullCalendar
                let calendar = new FullCalendar.Calendar(calendarEl, {
                  initialView: 'dayGridMonth',  // 預設為月視圖
                  headerToolbar: {              // 自訂工具列
                    left: 'prev,next today',    // 左側按鈕
                    center: 'title',            // 中央標題
                    right: 'dayGridMonth,timeGridWeek,timeGridDay'  // 右側切換按鈕
                  },
                  events: [                     // 加入一些事件
                    {
                      title: 'Meeting',
                      start: '2024-10-15T10:00:00',
                      end: '2024-10-15T12:00:00'
                    },
                    {
                      title: 'Conference',
                      start: '2024-10-17',
                      end: '2024-10-19'
                    },
                    {
                      title: 'Birthday Party',
                      start: '2024-10-30T19:00:00'
                    }
                  ],
                  editable: true,               // 啟用事件拖曳
                  selectable: true,             // 啟用區域選擇
                  select: function (info) {     // 當選擇區域時觸發
                    alert('Selected: ' + info.startStr + ' to ' + info.endStr);
                  }
                });
          
                // 渲染日曆
                calendar.render();
              });
            </script>
