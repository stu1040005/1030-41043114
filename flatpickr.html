日期選擇展示

flatpickr.js 官網

範例程式碼


            <!-- moment.js -->
            <script src="https://cdn.jsdelivr.net/npm/flatpickr"></script>
            <script>
            // 初始化 Flatpickr 日期選擇器
            flatpickr("#datepicker", {
            dateFormat: "Y-m-d", // 日期格式 (年-月-日)
            minDate: "today",    // 不允許選擇今天之前的日期
            defaultDate: "today", // 預設日期為今天
            enableTime: false     // 僅顯示日期，不顯示時間選擇
            });
            </script>
